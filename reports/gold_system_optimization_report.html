class GoldBacktester: """向量化回测引擎（无循环，防前视偏差）""" def backtest(self, signals: pd.Series, prices: pd.Series, transaction_cost=0.001, slippage=0.0005): # 关键：信号使用 shift(1) 防止前视偏差 position = signals.shift(1).fillna(0) returns = prices.pct_change() # 计算策略收益（扣除交易成本） trade_cost = position.diff().abs() * transaction_cost strategy_returns = position * returns - trade_cost equity = (1 + strategy_returns).cumprod() return { 'annual_return': strategy_returns.mean() * 252, 'sharpe': strategy_returns.mean() / strategy_returns.std() * np.sqrt(252), 'max_drawdown': (equity / equity.cummax() - 1).min(), 'calmar': (strategy_returns.mean() * 252) / abs((equity / equity.cummax() - 1).min()), 'win_rate': (strategy_returns > 0).mean() }

RSI 的 avg_gain/avg_loss 使用 .rolling().mean()（SMA），而 Wilder 原始定义使用指数移动平均（EMA，alpha=1/period）。SMA 在价格急剧波动后会低估/高估真实强弱，导致超买/超卖信号延迟约3-5根K线，在黄金单边行情中影响显著。同时 calculate_rsi 和 calculate_rsi_trend 两个方法代码完全重复，违反DRY原则。
🔧 修复方案
def _calculate_rsi_wilder(self, price_series, period=14): """Wilder标准RSI（EMA平滑，非SMA）""" delta = price_series.diff() gain = delta.clip(lower=0) loss = (-delta).clip(lower=0) # Wilder EMA: alpha = 1/period avg_gain = gain.ewm(alpha=1/period, adjust=False).mean() avg_loss = loss.ewm(alpha=1/period, adjust=False).mean() rs = avg_gain / avg_loss.replace(0, np.nan) rsi = 100 - (100 / (1 + rs)) return rsi

def calculate_position_size(self, signal, price, atr, risk_per_trade=0.02, # 单笔风险2%资金 target_volatility=0.10): # 目标年化波动率10% if signal == 0: return 0, 0, 0 # ATR动态止损（Chandelier Exit） stop_loss = price - signal * atr * 2.0 # 基于风险的头寸大小 risk_amount = abs(price - stop_loss) position_by_risk = risk_per_trade / (risk_amount / price) # 波动率目标约束 daily_vol = atr / price annual_vol = daily_vol * np.sqrt(252) position_by_vol = target_volatility / annual_vol # 取两者中较小的 final_position = min(position_by_risk, position_by_vol, 1.0) take_profit = price + signal * atr * 4.0 # 盈亏比 2:1 return final_position, stop_loss, take_profit
P2 缺少尾部风险量化指标（VaR / CVaR / Expected Shortfall）
涉及范围：src/report/gold_report_generator_v5.py 中的风险分析部分
报告中的"压力测试"仅展示百分比回撤（如-12%~-18%），未提供统计意义上的VaR/CVaR，缺少对极端尾部风险的量化。顶级机构（桥水/GS）的标准量化报告中VaR至少包括 95%/99% 置信水平的单日和月度VaR。
🔧 修复方案：添加参数VaR和历史模拟VaR
def calculate_risk_metrics(self, returns, confidence_levels=(0.95, 0.99)): """计算VaR和CVaR（历史模拟法）""" metrics = {} for cl in confidence_levels: var = np.percentile(returns, (1 - cl) * 100) cvar = returns[returns <= var].mean() # Expected Shortfall metrics[f'VaR_{int(cl*100)}'] = var metrics[f'CVaR_{int(cl*100)}'] = cvar # 最大回撤 equity = (1 + returns).cumprod() drawdown = equity / equity.cummax() - 1 metrics['max_drawdown'] = drawdown.min() metrics['calmar_ratio'] = returns.mean() * 252 / abs(drawdown.min()) return metrics

  • 信号阈值（1%收益率触发买入）从未验证

  • 策略间相关性未分析（可能高度同向，降低多样性）

  • 因子Zoo修剪：使用 Ridge / Lasso 回归替代 OLS，自动压缩无效因子权重

  • 机制自适应权重：ADX>25时趋势策略权重×1.5

📐 建议增加的回测指标

  • Sharpe Ratio（目标 > 1.2）
  • Sortino Ratio（目标 > 1.5）
